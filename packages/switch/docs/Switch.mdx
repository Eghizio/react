import { useState } from 'react';
import { Switch } from '../src';
import packageData from '../package.json';

# Switch

This component allows users to toggle between validity for a condition.

<PackageInfo packageData={packageData} componentNames={['Switch']} />

## Examples

```jsx example
function Example() {
    const [value, setValue] = useState(false);

    return (
        <Switch
            value={value}
            onClick={() => setValue(!value)}
            aria-label="Toggle switch"
        />
    );
}
```

#### To prevent the user from toggling the switch add the disabled prop

```jsx example
<Switch
    value={true}
    onClick={() => setValue(!value)}
    aria-label="Toggle switch"
    disabled
/>
```

#### Writing your own custom click handler

```jsx example
function Example() {
    const [value, setValue] = useState(false);

    return (
        <Switch
            value={value}
            onClick={() => {
                setValue(!value);
                alert('Your own custom action when toggled');
            }}
            aria-label="Toggle switch"
        />
    );
}
```

## Accessibility

The Switch needs either `aria-label` or `aria-labelledby` to be accessible to
screen readers.
